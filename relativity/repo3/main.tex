\subsection*{Q20.}
(5.3)式から
\begin{align}
  \begin{split}
    \eta_{00}=-1&=\sum_{\rho,\sigma}a^\rho_{\ 0}\eta_{\rho\sigma}a^\sigma_{\ 0}\\
    &=-(a^0_{\ 0})^2+(a^1_{\ 1})^2+(a^2_{\ 2})^2+(a^3_{\ 3})^2
  \end{split} 
\end{align}
したがって
\begin{align}
  (a^0_{\ 0})^2=1+(a^1_{\ 1})^2+(a^2_{\ 2})^2+(a^3_{\ 3})^2\geq 1
\end{align}
以上から
\begin{align}
  |a^0_{\ 0}|\geq 1
\end{align}
